         src="https://cdn.pixabay.com/download/audio/2022/10/25/audio_946bc4be4e.mp3"></audio>

<script>
/* ---------- AUDIO ---------- */
const audio = document.getElementById('audio');
const playBtn = document.getElementById('play');
const fileIn = document.getElementById('file');
const vol = document.getElementById('vol');

let ctx, analyser, srcNode, freq;
function ensureAudio(){
  if(ctx) return;
  ctx = new (window.AudioContext||window.webkitAudioContext)();
  analyser = ctx.createAnalyser();
  analyser.fftSize = 1024;
  analyser.smoothingTimeConstant = 0.75;
  srcNode = ctx.createMediaElementSource(audio);
  srcNode.connect(analyser);
  analyser.connect(ctx.destination);
  freq = new Uint8Array(analyser.frequencyBinCount);
}
playBtn.onclick = async () => {
  ensureAudio();
  if(ctx.state === 'suspended') await ctx.resume();
  if(audio.paused){ await audio.play(); playBtn.textContent='❚❚ Pause'; }
  else { audio.pause(); playBtn.textContent='▶ Play'; }
};
fileIn.onchange = e => {
  const f = e.target.files[0]; if(!f) return;
  audio.src = URL.createObjectURL(f);
  audio.play().then(()=>{ensureAudio(); playBtn.textContent='❚❚ Pause';});
};
vol.oninput = () => audio.volume = +vol.value;
audio.volume = +vol.value;

/* ---------- CANVAS ---------- */
const cvs = document.getElementById('bg');
const c = cvs.getContext('2d');
let W, H;
function resize(){
  const dpr = Math.min(window.devicePixelRatio||1, 2);
  W = cvs.clientWidth; H = cvs.clientHeight;
  cvs.width = W*dpr; cvs.height = H*dpr;
  c.setTransform(dpr,0,0,dpr,0,0);
}
addEventListener('resize', resize); resize();

const stars = Array.from({length:160},()=>({
  x:Math.random(), y:Math.random()*0.55,
  s:Math.random()*1.6+0.3, tw:Math.random()
}));

let roadOffset = 0, lastBeat = 0, beatFlash = 0, prevBass = 0;

function frame(t){
  let bass=0, mid=0, treble=0;
  if(analyser){
    analyser.getByteFrequencyData(freq);
    const n = freq.length;
    const avg = (a,b)=>{ let s=0; for(let i=a;i<b;i++) s+=freq[i]; return s/((b-a)*255); };
    bass   = avg(0, Math.floor(n*0.08));
    mid    = avg(Math.floor(n*0.08), Math.floor(n*0.4));
    treble = avg(Math.floor(n*0.4), n);
  }
  // beat detection (bass jump)
  if(bass - prevBass > 0.18 && t - lastBeat > 180){ lastBeat=t; beatFlash=1; }
  prevBass = bass;
  beatFlash *= 0.9;

  /* sky */
  const sky = c.createLinearGradient(0,0,0,H*0.65);
  sky.addColorStop(0,'#1a0033');
  sky.addColorStop(1,'#ff3cac');
  c.fillStyle = sky;
  c.fillRect(0,0,W,H*0.65);

  /* stars */
  for(const s of stars){
    const tw = 0.5+0.5*Math.sin(t*0.003+s.tw*10);
    const a = Math.min(1, 0.35+tw*0.4+treble*1.2);
    c.fillStyle = `rgba(255,255,255,${a})`;
    const sz = s.s*(1+treble*2);
    c.fillRect(s.x*W, s.y*H, sz, sz);
  }

  /* sun */
  const sunR = H*0.18*(1+bass*0.15);
  const cx = W*0.5, cy = H*0.5;
  // glow
  const glow = c.createRadialGradient(cx,cy,sunR*0.8,cx,cy,sunR*2.6);
  glow.addColorStop(0,'rgba(255,180,80,0.45)');
  glow.addColorStop(0.5,'rgba(255,80,140,0.18)');
  glow.addColorStop(1,'rgba(0,0,0,0)');
  c.fillStyle = glow;
  c.fillRect(cx-sunR*3, cy-sunR*3, sunR*6, sunR*6);
  // sun body
  const sg = c.createLinearGradient(0,cy-sunR,0,cy+sunR);
  sg.addColorStop(0,'#ffd86b');
  sg.addColorStop(0.55,'#ff8a3d');
  sg.addColorStop(1,'#ff3cac');
  c.fillStyle = sg;
  c.beginPath(); c.arc(cx,cy,sunR,0,Math.PI*2); c.fill();
  // stripes through sun
  c.globalCompositeOperation = 'destination-out';
  for(let i=0;i<9;i++){
    const y = cy + sunR*0.25 + i*7;
    c.fillRect(cx-sunR, y, sunR*2, 2+i*0.9);
  }
  c.globalCompositeOperation = 'source-over';

  /* road */
  c.fillStyle = '#0a001a';
  c.fillRect(0, H*0.65, W, H*0.35);
  const horizon = H*0.65;
  roadOffset += 2 + bass*14 + mid*4;

  // horizontal lines
  c.strokeStyle = '#ff3cac';
  c.shadowColor = '#ff3cac';
  c.shadowBlur  = 12 + bass*30;
  c.globalAlpha = 0.45 + bass*0.5;
  c.lineWidth   = 1.5;
  for(let i=0;i<18;i++){
    const p = ((i*60 + roadOffset) % 1080)/1080;
    const y = horizon + Math.pow(p,2.2)*(H-horizon);
    const k = (y-horizon)/(H-horizon);
    c.beginPath();
    c.moveTo(W*0.5 - k*W*0.9, y);
    c.lineTo(W*0.5 + k*W*0.9, y);
    c.stroke();
  }
  // vertical lines (perspective)
  for(let i=-10;i<=10;i++){
    c.beginPath();
    c.moveTo(W*0.5 + i*12, horizon);
    c.lineTo(W*0.5 + i*W*0.18, H);
    c.stroke();
  }
  c.globalAlpha = 1; c.shadowBlur = 0;

  /* equalizer bars on top */
  if(analyser){
    const bars = 48;
    const bw = (W*0.5)/bars;
    c.save();
    c.fillStyle = '#ff3cac';
    c.shadowColor = '#ff3cac';
    c.shadowBlur = 8;
    for(let i=0;i<bars;i++){
      const v = freq[Math.floor((i/bars)*freq.length*0.6)]/255;
      const h = v*80;
      c.fillRect(W*0.25 + i*bw, 30, bw-2, h);
    }
    c.restore();
  }

  /* beat flash */
  if(beatFlash > 0.05){
    c.fillStyle = `rgba(255,255,255,${beatFlash*0.25})`;
    c.fillRect(0,0,W,H);
  }

  requestAnimationFrame(frame);
}
requestAnimationFrame(frame);
</script>
